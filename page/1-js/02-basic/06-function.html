함수 관련

여러 항목을 반환할때

return
  (some + long + whatever * f(a) + f(b)) // 이거는 안됨
return; // 이렇게 인식됨
  (some + long + whatever * f(a) + f(b)) // 이거는 안됨
return (
  some + long + whatever * f(a) + f(b)
  ) // 이렇게 해야함

함수 이름짓기

  • get.. : 값을 반환함
  • calc... : 계산함
  • create... : 생성함
  • check... , is... : 무언가를 확인하고 불린값을 반환함
  • show... : 메세지를 보여줌

함수는 동작 하나만 담당해야 합니다.

function showPrimes(n) {
  for (let i = 2; i < n; i++) {
    if (!isPrime(i)) continue;
    alert(i);  // a prime
  }
}
function isPrime(n) {
  for (let i = 2; i < n; i++) {
    if ( n % i == 0) return false;
  }
  return true;
}
function ask(question, yes, no) {
  if (confirm(question)) yes()
  else no();
}
function showOk() { alert( "동의하셨습니다." ); }
function showCancel() { alert( "취소 버튼을 누르셨습니다." ); }
// 사용법: 함수 showOk와 showCancel가 ask 함수의 인수로 전달됨
ask("동의하십니까?", showOk, showCancel);

함수 ask의 인수, showOk와 showCancel을 콜랙함수 또는 콜백이라고 함

함수를 함수의 인수로 전달하고, 필요하다면 인수로 전달된 함수를 나중에 호출(call back)하는 것이 콜백함수의 개념
